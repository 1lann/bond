143  void start()
144  {
145  BOOST_ASSERT(_cq);
146 
147  if (_threads.empty())
148  {
149  _threads.reserve(_numThreads);
150 
151  for (size_t i = 0; i < _numThreads; ++i)
152  {
153  _threads.emplace_back([this]()
154  {
155  void* tag;
156  bool ok;
157  while (_cq->Next(&tag, &ok))
158  {
159  BOOST_ASSERT(tag);
160  static_cast<detail::io_manager_tag*>(tag)->invoke(ok);
161  }
162  });
163  }
164  }
165  }

194  void wait()
195  {
196  bool shouldShutdown = !_isShutdownInProgress.test_and_set();
197  if (shouldShutdown)
198  {
199  // borrow the current thread to clean up
200  for (auto& thread : _threads)
201  {
202  BOOST_ASSERT(thread.joinable());
203  thread.join();
204  }
205 
206  _threads.clear();
207 
208  _cq.reset();
209  _shutdownCompleted.set();
210  }
211  else
212  {
213  // some other thread is performing clean up, so wait for it
214  _shutdownCompleted.wait();
215  }
216  }

219  static size_t compute_real_num_threads(size_t numThreads)
220  {
221  if (numThreads == USE_HARDWARE_CONC)
222  {
223  numThreads = static_cast<size_t>(std::thread::hardware_concurrency());
224  }
225 
226  // hardware_concurency can fail and return 0. If so, we need a
227  // non-zero number of threads.
228  const size_t recourseNumThreads = 2;
229  return numThreads != 0 ? numThreads : recourseNumThreads;
230  }
