129  void start()
130  {
131  BOOST_ASSERT(_cq);
132 
133  if (_threads.empty())
134  {
135  _threads.reserve(_numThreads);
136 
137  for (size_t i = 0; i < _numThreads; ++i)
138  {
139  _threads.emplace_back([this]()
140  {
141  void* tag;
142  bool ok;
143  while (_cq->Next(&tag, &ok))
144  {
145  BOOST_ASSERT(tag);
146  static_cast<detail::io_manager_tag*>(tag)->invoke(ok);
147  }
148  });
149  }
150  }
151  }
