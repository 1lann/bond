174  server BuildAndStart()
175  {
176  BOOST_STATIC_ASSERT(std::is_move_constructible<server>::value);
177 
178  if (!_scheduler)
179  {
180  _scheduler = thread_pool{};
181  }
182 
183  auto cq = _builder.AddCompletionQueue();
184  auto grpcServer = _builder.BuildAndStart();
185 
186  for (auto& service : _services)
187  {
188  service->start(cq.get(), _scheduler);
189  }
190 
191  std::unique_ptr<io_manager> ioManager{
192  new io_manager{
193  std::thread::hardware_concurrency(),
194  /*delay=*/ false,
195  std::move(cq) } };
196 
197  return server{ std::move(grpcServer), std::move(ioManager) };
198  }
