181  std::unique_ptr<bond::ext::gRPC::server_core<TThreadPool>> BuildAndStart()
182  {
183  std::unique_ptr<grpc::ServerCompletionQueue> cq =
184  _grpcServerBuilder.AddCompletionQueue();
185  std::unique_ptr<grpc::Server> server =
186  _grpcServerBuilder.BuildAndStart();
187 
188  if (!_threadPool)
189  {
190  _threadPool = std::make_shared<TThreadPool>();
191  }
192 
193  // Tickle all the services so they queue a receive for all their
194  // methods.
195  for (auto& service : _services)
196  {
197  service->start(cq.get(), _threadPool);
198  }
199 
200  std::unique_ptr<bond::ext::gRPC::server_core<TThreadPool>> result {
201  new bond::ext::gRPC::server_core<TThreadPool> {
202  std::move(server),
203  std::move(cq) } };
204 
205  return result;
206  }
207 
208  private:

210  std::set<detail::service<TThreadPool>*> _services;
211  std::shared_ptr<TThreadPool> _threadPool;
