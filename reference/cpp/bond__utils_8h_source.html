39  grpc_byte_buffer** bp,
40  bool* own_buffer)
41  {
42  *own_buffer = true;

44  bond::OutputBuffer output;

58  }

60  static Status Deserialize(grpc_byte_buffer* buffer, bond::bonded<T>* msg)
61  {
62  if (!buffer)
63  {
64  return Status(StatusCode::INTERNAL, "No payload");
65  }
66 
67  const size_t bufferSize = grpc_byte_buffer_length(buffer);
68 
69  if (bufferSize > (std::numeric_limits<uint32_t>::max)())
70  {
71  grpc_byte_buffer_destroy(buffer);
72  return Status(StatusCode::INTERNAL, "Buffer is too large");
73  }
74 
75  boost::shared_ptr<char[]> buff = boost::make_shared_noinit<char[]>(bufferSize);
76 
77  // TODO: exception safety of reader, buffer
78  grpc_byte_buffer_reader reader;
79  if (!grpc_byte_buffer_reader_init(&reader, buffer))
80  {
81  grpc_byte_buffer_destroy(buffer);
82  return Status(StatusCode::INTERNAL, "Failed to init buffer reader");
83  }
84 
85  char* dest = buff.get();
86 
87  for (grpc_slice grpc_s; grpc_byte_buffer_reader_next(&reader, &grpc_s) != 0;)
88  {
89  grpc::Slice s{ grpc_s, grpc::Slice::STEAL_REF };
90  std::memcpy(dest, s.begin(), s.size());
91  dest += s.size();
92  }
93 
94  BOOST_ASSERT(dest == buff.get() + bufferSize);
95 
96  grpc_byte_buffer_reader_destroy(&reader);
97  grpc_byte_buffer_destroy(buffer);
98 
99  // TODO: create a Bond input stream over grpc_byte_buffer to avoid
100  // having to make this copy into a blob.
101  bond::blob data = bond::blob(buff, static_cast<uint32_t>(bufferSize));
102  bond::CompactBinaryReader<bond::InputBuffer> cbreader(data);
103  *msg = bond::bonded<T>(cbreader);
104 
105  return Status::OK;
106  }
107  };
