89  void Finish(const bond::bonded<Response>& msg, const grpc::Status& status)
90  {
91  bool wasResponseSent = _responseSentFlag.test_and_set();
92  if (!wasResponseSent)
93  {
94  _responder.Finish(msg, status, tag());
95  }
96  }
97 
98  void FinishWithError(const grpc::Status& status)
99  {
100  bool wasResponseSent = _responseSentFlag.test_and_set();
101  if (!wasResponseSent)
102  {
103  _responder.FinishWithError(status, tag());
104  }
105  }
106 
107  private:
108  void invoke(bool /* ok */) override
109  {
110  // The response has been sent, so we no longer need to keep
111  // ourselves alive: release the implicit initial refcount that
112  // this instance was constructed with.
113  Release();
114  }
115 
116  void AddRef() noexcept
117  {
118  _refCount.fetch_add(1, std::memory_order_relaxed);
119  }
120 
121  void Release()
122  {
123  if (_refCount.fetch_sub(1, std::memory_order_acq_rel) == 1)
124  {
125  delete this;
126  }
127  }
128 
129  void TryFinishWithError()
130  {
131  if (_refCount.load(std::memory_order::memory_order_acquire) == 2)
132  {
133  // The last user reference has just gone away, but Finish was
134  // not called. In this case, we are responsible for sending
135  // an error response and decrementing the final ref
136  // count. FinishWithError will schedule the send of the
137  // error response, and notification of completion of the
138  // send via invoke() will decrement the final ref count.
139  // Since we hold the ref count ourselves, we will not
140  // get deleted until we're done sending.
141  //
142  // Even when multiple threads enter this case, at most one
143  // will succeed in sending error, thanks to _responseSentFlag.
144 
145  FinishWithError({ grpc::StatusCode::INTERNAL, "An internal server error has occurred." });
146  }
147  }
